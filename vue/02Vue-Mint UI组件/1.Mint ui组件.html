Mint UI 基于 Vue.js的移动端组件库   
    
    1.安装：
        #Vue 1.x
        npm install mint-ui@1 -S 生产环境依赖
        #Vue 2.0   
        npm install mint-ui -S 生产环境依赖

    2.引入全部组件
    import Vue from 'vue'
    import Mint from 'mint-ui' //把所有组件导入
    Vue.use(Mint) //注册所有组件到全局  

    3.按需引入组件
    import {Cell,Checklist} from 'mint-ui'
    Vue.component(Cell.name,Cell)  //注册组件到全局
    Vue.component(Checklist.name,Checklist)
